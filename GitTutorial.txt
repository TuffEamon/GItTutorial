using System;

class Program
{
    static void Main()
    {
        int sum = 0;
        Console.WriteLine("Counting 1 to 10:");

        for (int i = 1; i <= 10; i++)
        {
            sum += i;

            if (i % 2 == 0)
                Console.WriteLine(i + " is even");
            else
                Console.WriteLine(i + " is odd");
        }

        Console.WriteLine("Total sum = " + sum);
        Console.WriteLine("Done!");
    }
}
